set fileformats=dos,unix,mac

set shiftwidth=4

set guioptions-=T

"
set nobackup
set noundofile
"クリップボードをWindowsと連携
source $VIMRUNTIME/mswin.vim

"自動改行機能Off
set textwidth=0
set formatoptions=q

function! GetStatusEx()
    let str = ''
    let str = str . '[' . &fileformat . ']'
    if has('multi_byte') && &fileencoding != ''
        let str = '[' . &fenc . ']' . str
    endif
    return str
endfunction
set statusline=%y%{GetStatusEx()}%F%m%r=<%c:%l>
