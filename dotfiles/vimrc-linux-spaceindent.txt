" for Linux
":syntax off
colorscheme desert
"
set encoding=utf-8
set fileencoding=
set fileencodings=iso-2022-jp,utf-8,cp932,euc-jp
set fileformats=unix,dos,mac
"
set shiftwidth=4
set tabstop=4
set autoindent
set expandtab
set softtabstop=0
set smartindent
"
set nobackup
set ignorecase
"
set ambiwidth=double
set hlsearch
"
set laststatus=2
function! GetStatusEx()
    let str = ''
    let str = str . '[' . &fileformat . ']'
    if has('multi_byte') && &fileencoding != ''
        let str = '[' . &fenc . ']' . str
    endif
    return str
endfunction
set statusline=%y%{GetStatusEx()}%F%m%r=<%c:%l>
